var totalBotoes = 11;
var botoesClicados = 0;

function verificarBotoesClicados() {
    if (botoesClicados === totalBotoes) {
        document.getElementsByClassName('continue')[0].style.display = 'block';
    }
}


function trocar1(classeBotao) {
    // Oculta todas as imagens
    var imagens = document.querySelectorAll('img.b1');
    imagens.forEach(function(imagem) {
        imagem.style.display = 'block';
    });
    botoesClicados++;
    
    verificarBotoesClicados();
}
